\chapter{Human-Centered Evaluation of Toolset and Generated Results}\label{ch:evaluation}

\section{Motivation for Human Evaluation}
\begin{itemize}
    \item Limitations of automatic image quality metrics and software tool evaluations.
    \item Relevance of human perception for semantic interpretation.
\end{itemize}

\section{User Study Design}
\begin{itemize}
    \item Participant selection (e.g. design or media-related courses).
    \item Structure of the survey (e.g. Likert scales, comparative judgments).
    \item Evaluation criteria for both generated results and tool interaction (e.g. coherence, controllability, perceived continuity, usability).
\end{itemize}

\section{Evaluation Results}
\begin{itemize}
    \item Aggregated results of the survey.
    \item Observed trends and variances in perception and interaction.
    \item Implications for interpretability, controllability, and usability of latent space navigation.
\end{itemize}
